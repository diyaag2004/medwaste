const authController = async(req,res) =>{
try {
  const user = await userModel.findOne({_id: req.body.userId});
  if(!user){
    return res.status(200).send({
      message:"user not found",
      success:false,
    });
  }
  else{
    res.status(200).send({
      success:true,
      data:{
        name:user.name,
        email:user.email,
      },

    });
  }
}

 catch(error){
  console.log(error);
  res.status(500).send({
    message:"auth error",
    success:false,
    error,
  });
}
}
